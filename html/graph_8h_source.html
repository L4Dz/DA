484template <class T>
485bool findAugmentingPath(Graph<T> *g, Vertex<T> *s, Vertex<T> *t) {
486 for (auto v : g->getVertexSet()) v->setVisited(false);
487 s->setVisited(true);
488 std::queue<Vertex<T>*> q;
489 q.push(s);
490
491 while (!q.empty() && !t->isVisited()) {
492 auto v = q.front(); q.pop();
493 for (auto e : v->getAdj())
494 testAndVisit(q, e, e->getDest(), e->getWeight() - e->getFlow());
495 for (auto e : v->getIncoming())
496 testAndVisit(q, e, e->getOrig(), e->getFlow());
497 }
498 return t->isVisited();
499}

531template <class T>
532void edmondsKarp(Graph<T> *g, T source, T target) {
533 Vertex<T>* s = g->findVertex(source);
534 Vertex<T>* t = g->findVertex(target);
535 if (s == nullptr || t == nullptr || s == t) return;
536
537 for (auto v : g->getVertexSet())
538 for (auto e : v->getAdj()) e->setFlow(0);
539
540 while (findAugmentingPath(g, s, t)) {
541 double f = INF;
542 auto v = t;
543 while (v != s) {
544 auto e = v->getPath();
545 if (e->getDest() == v) {
546 f = std::min(f, e->getWeight() - e->getFlow());
547 v = e->getOrig();
548 } else {
549 f = std::min(f, e->getFlow());
550 v = e->getDest();
551 }
552 }
553 v = t;
554 while (v != s) {
555 auto e = v->getPath();
556 if (e->getDest() == v) {
557 e->setFlow(e->getFlow() + f);
558 v = e->getOrig();
559 } else {
560 e->setFlow(e->getFlow() - f);
561 v = e->getDest();
562 }
563 }
564 }

590 Edge<T> *edge = *it;
