164 auto getMatchedDomain = [this](auto sub, auto rev) -> int {
165 if (sub->primaryDomain == rev->primaryDomain) return sub->primaryDomain;
166 if (params.useSecondaryRev && sub->primaryDomain == rev->secondaryDomain) return sub->primaryDomain;
167 if (params.useSecondarySub && sub->secondaryDomain == rev->primaryDomain) return sub->secondaryDomain;
168 if (params.useSecondarySub && params.useSecondaryRev && sub->secondaryDomain == rev->secondaryDomain && sub->secondaryDomain != -1) return sub->secondaryDomain;
169 return -1;
170 };
171
172 // 1. Sub -> Rev
173 outFile << "#SubmissionId,ReviewerId,Match\n";
174 int total = 0;
175 std::vector<std::tuple<int, int, int>> revMap;
176 for (auto v : g.getVertexSet()) {
177 if (v->type == VertexType::SUBMISSION) {
178 for (auto e : v->getAdj()) {
179 if (e->getFlow() > 0 && e->getDest()->type == VertexType::REVIEWER) {
180 int m = getMatchedDomain(v, e->getDest());
181 outFile << v->getInfo() << ", " << (e->getDest()->getInfo() - 1000) << ", " << m << "\n";
182 revMap.push_back({e->getDest()->getInfo() - 1000, v->getInfo(), m});
183 total++;
184 }
185 }
186 }
187 }
188
189 // 2. Rev -> Sub (Sorted)
190 outFile << "#ReviewerId,SubmissionId,Match\n";
191 std::sort(revMap.begin(), revMap.end());
192 for (auto const& [rid, sid, m] : revMap) outFile << rid << ", " << sid << ", " << m << "\n";
193
194 // 3. Total e Risco
195 outFile << "#Total: " << total << "\n";
196 if (control.riskAnalysis > 0) {
197 outFile << "#Risk Analysis: " << control.riskAnalysis << "\n";
198 std::vector<int> risky = findRiskyReviewers();
199 for (size_t i = 0; i < risky.size(); ++i) {
200 outFile << risky[i] << (i < risky.size() - 1 ? ", " : "");
201 }
202 outFile << "\n";
203 }
204
205 // 4. Missing Reviews
206 bool first = true;
207 for (auto v : g.getVertexSet()) {
208 if (v->type == VertexType::SUBMISSION) {
209 int count = 0;
210 for (auto e : v->getAdj()) if (e->getFlow() > 0 && e->getDest()->type == VertexType::REVIEWER) count++;
211 if (count < params.minReviews) {
212 if (first) { outFile << "#SubmissionId,Domain,MissingReviews\n"; first = false; }
213 outFile << v->getInfo() << ", " << v->primaryDomain << ", " << (params.minReviews - count) << "\n";
214 }
215 }
