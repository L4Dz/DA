216 void loadData(const std::string &filename, Graph<int> &g, AssignmentParams &params, ControlSettings &control) {
217 std::ifstream file(filename);
218 if (!file.is_open()) {
219 std::cerr << "Error: Could not open file " << filename << std::endl;
220 return;
221 }
222
223 std::string line, section = "NONE";
224
225 while (std::getline(file, line)) {
226 line = trim(line);
227 if (line.empty()) continue;
228
229 // 1. Detect Section Headers
230 if (line[0] == '#') {
231 std::string header = trim(line.substr(1));
232 if (header.empty()) continue;
233
234 // Skip column label lines (Id, Name, Title, ...)
235 if (header.find("Id") == 0 || header.find("Name") == 0 || header.find("Title") == 0) continue;
236
237 // Set the current section context
238 if (header.find("Submissions") != std::string::npos) section = "SUBMISSIONS";
239 else if (header.find("Reviewers") != std::string::npos) section = "REVIEWERS";
240 else if (header.find("Parameters") != std::string::npos) section = "PARAMETERS";
241 else if (header.find("Control") != std::string::npos) section = "CONTROL";
242 continue;
243 }
244
245 // 2. Split Data Row
246 std::stringstream ss(line);
247 std::string cell;
248 std::vector<std::string> row;
249 while (std::getline(ss, cell, ',')) {
250 row.push_back(trim(cell));
251 }
252
253 // 3. Process based on Section
254 try {
255 if (section == "SUBMISSIONS" && row.size() >= 5) {
256 int id = std::stoi(row[0]);
257 g.addVertex(id);
258 auto v = g.findVertex(id);
259 v->type = VertexType::SUBMISSION;
260 v->title = row[1];
261 v->primaryDomain = std::stoi(row[4]);
262 v->secondaryDomain = (row.size() > 5 && !row[5].empty()) ? std::stoi(row[5]) : -1;
263 }
264 else if (section == "REVIEWERS" && row.size() >= 4) {
265 int id = std::stoi(row[0]) + 1000;
266 g.addVertex(id);
267 auto v = g.findVertex(id);
268 v->type = VertexType::REVIEWER;
269 v->title = row[1];
270 v->primaryDomain = std::stoi(row[3]);
271 v->secondaryDomain = (row.size() > 4 && !row[4].empty()) ? std::stoi(row[4]) : -1;
272 }
273 else if (section == "PARAMETERS" && row.size() >= 2) {
274 std::string key = row[0];
275 int val = std::stoi(row[1]);
276 if (key == "MinReviewsPerSubmission") params.minReviews = val;
277 else if (key == "MaxReviewsPerReviewer") params.maxReviews = val;
278 else if (key == "PrimaryReviewerExpertise") params.usePrimaryRev = val;
279 else if (key == "SecondaryReviewerExpertise") params.useSecondaryRev = val;
280 else if (key == "PrimarySubmissionDomain") params.usePrimarySub = val;
281 else if (key == "SecondarySubmissionDomain") params.useSecondarySub = val;
282 }
283 else if (section == "CONTROL" && row.size() >= 2) {
284 std::string key = row[0];
285 if (key == "GenerateAssignments") control.generateAssignments = std::stoi(row[1]);
286 else if (key == "RiskAnalysis") control.riskAnalysis = std::stoi(row[1]);
287 else if (key == "OutputFileName") control.outputFileName = row[1];
288 }
289 } catch (...) {
290 // Ignore rows that don't match the expected format
291 // (PT) Ignora linhas que não correspondem ao formato esperado
292 continue;
293 }
294 }
295 file.close();

331 void buildNetwork(Graph<int> &g, const AssignmentParams &p) {
332 // 1. Add super-source (0) and super-sink (9999)
333 // (PT) Adiciona super-fonte (0) e super-sumidouro (9999)
334 g.addVertex(0);
335 g.addVertex(9999);
336 auto s = g.findVertex(0);
337 auto t = g.findVertex(9999);
338 s->type = VertexType::SOURCE;
339 t->type = VertexType::SINK;
340
341 auto nodes = g.getVertexSet();
342 for (auto v : nodes) {
343 if (v->type == VertexType::SUBMISSION) {
344 // Each paper needs 'minReviews' reviews → source edge capacity
345 // (PT) Cada artigo precisa de 'minReviews' revisões → capacidade da aresta fonte
346 s->addEdge(v, p.minReviews);
347
348 for (auto rev : nodes) {
349 if (rev->type == VertexType::REVIEWER) {
350 bool isMatch = false;
351
352 // Case A: Primary to Primary (gold standard match)
353 // (PT) Caso A: Primário para Primário (correspondência ideal)
354 if (v->primaryDomain == rev->primaryDomain) {
355 isMatch = true;
356 }
357 // Case B: Primary submission domain vs Secondary reviewer domain
358 // (PT) Caso B: Domínio primário da submissão vs domínio secundário do revisor
359 else if (p.useSecondaryRev && v->primaryDomain == rev->secondaryDomain) {
360 isMatch = true;
361 }
362 // Case C: Secondary submission domain vs Primary reviewer domain
363 // (PT) Caso C: Domínio secundário da submissão vs domínio primário do revisor
364 else if (p.useSecondarySub && v->secondaryDomain == rev->primaryDomain) {
365 isMatch = true;
366 }
367 // Case D: Secondary submission domain vs Secondary reviewer domain
368 // (PT) Caso D: Domínio secundário da submissão vs domínio secundário do revisor
369 else if (p.useSecondarySub && p.useSecondaryRev &&
370 v->secondaryDomain == rev->secondaryDomain && v->secondaryDomain != -1) {
371 isMatch = true;
372 }
373
374 if (isMatch) {
375 // Capacity = 1: one reviewer handles one paper at a time
376 // (PT) Capacidade = 1: um revisor trata de um artigo de cada vez
377 v->addEdge(rev, 1);
378 }
379 }
380 }
381 }
382 else if (v->type == VertexType::REVIEWER) {
383 // Reviewer workload cap: limits total submissions assigned
384 // (PT) Limite de carga do revisor: limita o total de submissões atribuídas
385 v->addEdge(t, p.maxReviews);
386 }
387 }
388 }
